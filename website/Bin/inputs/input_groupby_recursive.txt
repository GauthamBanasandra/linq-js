function linq_query()
{
	var fruits = ['apple', 'mango', 'banana', 'grapes'];
	var new_fruits =
	    from fruit in
	        (
	            from fr in
	                (
                        from fru in fruits
                        select fru
                    )
	            where {fr.length <= 5}
	            select fr
            )
        where {fruit.length == 5}
        group fruit by fruit.length into nfl
        select nfl;

	console.log(new_fruits);
}
linq_query();
